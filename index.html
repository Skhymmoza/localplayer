import java.net.http.*;
import org.json.JSONObject;

public class LocalPlayerAPI {
    private static final String API_KEY = "LP-a1b2c3d4f5h6";
    private static final String BASE_URL = 
        "https://localplayer.vercel.app/api";
    
    public static JSONObject connect() {
        HttpClient client = HttpClient.newHttpClient();
        JSONObject body = new JSONObject();
        body.put("device_id", "android_001");
        body.put("network_uuid", "550e8400-e29b");
        
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(BASE_URL + "/" + API_KEY + "/connect"))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers
                .ofString(body.toString()))
            .build();
        
        return new JSONObject(response.body());
    }
}

public static JSONObject getDevices() {
    HttpClient client = HttpClient.newHttpClient();
    
    HttpRequest request = HttpRequest.newBuilder()
        .uri(URI.create(BASE_URL + "/" + API_KEY + "/devices"))
        .GET()
        .build();
    
    HttpResponse response = client.send(
        request, HttpResponse.BodyHandlers.ofString()
    );
    
    return new JSONObject(response.body());
}

public static void disconnect(String deviceId) {
    JSONObject body = new JSONObject();
    body.put("device_id", deviceId);
    
    HttpRequest request = HttpRequest.newBuilder()
        .uri(URI.create(BASE_URL + "/" + API_KEY + "/disconnect"))
        .POST(HttpRequest.BodyPublishers
            .ofString(body.toString()))
        .build();
    
    client.send(request, 
        HttpResponse.BodyHandlers.ofString());
}

const API_KEY = 'LP-a1b2c3d4f5h6';
const BASE_URL = 'https://localplayer.vercel.app/api';

async function connectToNetwork() {
    const response = await fetch(
        `${BASE_URL}/${API_KEY}/connect`,
        {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                device_id: 'android_001',
                network_uuid: '550e8400-e29b'
            })
        }
    );
    return await response.json();
}

connectToNetwork().then(result => {
    console.log(`IP: ${result.assigned_ip}`);
});

    const response = await fetch(
        `${BASE_URL}/${API_KEY}/devices`

    return await response.json();
}

getDevices().then(result => {

        console.log(`${device.name}: ${device.ip}`);
    });

POST

/api/{API_KEY}/disconnect

Отключение устройства от сети

Параметры: device_id
GET

/api/{API_KEY}/status
