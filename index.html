Простая интеграция

Подключите устройство к VPN сети всего за несколько строк кода

Быстрое подключение 
import requests

API_KEY = "LP-a1b2c3d4f5h6"
BASE_URL = "https://localplayer-tau.vercel.app/api"

# Подключение к сети
response = requests.post(
    f"{BASE_URL}/{API_KEY}/connect",
    json={
        "device_id": "android_001",
        "network_uuid": "550e8400-e29b"
    }
)

result = response.json()
print(f"Подключено! IP: {result['assigned_ip']}")
Быстрое подключение 
import java.net.http.*;
import org.json.JSONObject;

String API_KEY = "LP-a1b2c3d4f5h6";
String BASE_URL = "https://localplayer-tau.vercel.app/api";

HttpClient client = HttpClient.newHttpClient();
JSONObject body = new JSONObject()
    .put("device_id", "android_001")
    .put("network_uuid", "550e8400-e29b");

HttpRequest request = HttpRequest.newBuilder()
    .uri(URI.create(BASE_URL + "/" + API_KEY + "/connect"))
    .POST(HttpRequest.BodyPublishers.ofString(body.toString()))
    .build();

HttpResponse response = client.send(request,
    HttpResponse.BodyHandlers.ofString());
System.out.println("Подключено!");

const API_KEY = 'LP-a1b2c3d4f5h6';
const BASE_URL = 'https://localplayer-tau.vercel.app/api';

// Подключение к сети
const response = await fetch(`${BASE_URL}/${API_KEY}/connect`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        device_id: 'android_001',
        network_uuid: '550e8400-e29b'
    })

const result = await response.json();
console.log(`Подключено! IP: ${result.assigned_ip}`);
